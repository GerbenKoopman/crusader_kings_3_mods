aqoa_council_salaries = {

		flag = aqoa_council_salaries_none

		can_keep = {
			is_landed = yes
			highest_held_title_tier >= tier_county
	    }

			councillor_opinion = -15
			glory_hound_opinion = -15
			courtly_opinion = -15

			monthly_income_mult = 0.3	#5 councillors x 0.04 = 0.1, avg 5 knights x 0.02 = 0.05
		}

		pass_cost = {
			prestige = 50
		}

		can_pass = {
			trigger_if = {
				limit = {
					is_ruler = yes
				}
				custom_description = {
					subject = root
					text = "has_councillor_salary_policy_cooldown"
					NOT = {
						has_variable = councillor_salary_policy_cooldown
					}
				}
			}
		}

		on_pass = {
			# Set regular cooldown.
			set_variable = {
				name = councillor_salary_policy_cooldown
				years = 2
			}
		}

		ai_will_do = {
			value = 0
		}
	}
