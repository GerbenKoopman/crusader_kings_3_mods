# This trigger checks if a character is a "Manorial Lord"
# usage: is_manorial_lord = yes
is_manorial_lord = {
	# 1. Must have a domicile (The estate mechanic)
	has_domicile = yes
	# 2. That Domicile MUST be a Manor
	OR = {
		# The Modded Feudal Manor (From mod 3624056113)
		domicile = { domicile_type = feudalmanor }
		
		# The Vanilla Admin Manor (Roads to Power compatibility)
		domicile = { domicile_type = manor_domicile }
	}
}
