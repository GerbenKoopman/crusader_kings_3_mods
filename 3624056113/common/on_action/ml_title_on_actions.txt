on_title_gain = {
    on_actions = {
        gain_landed_title_on_action
    }
}

gain_landed_title_on_action = {
    effect = {
        if = {
            limit = {
				NOT = { 
				    government_has_flag = government_uses_domicile_but_not_adventurer
				}
			    any_held_title = {
				    is_noble_family_title = no
			    }
            }

			random_held_title = {
				limit = {
					is_noble_family_title = yes
				}
				save_scope_as = title_gnt_destroy
				root = {
					destroy_title = scope:title_gnt_destroy
				}
			}
        }
    }
}
